public class Rectangulo{
    //atributos
    int alto;
    int ancho; 
    boolean relleno;
    //métodos 
    public int devolverArea () {
        return alto*ancho;
    }
    public void rellenar(boolean r){
        relleno=r;
    }
    public void cambiarTamano(int an, int al){
        ancho=an;
        alto=al;
    }
    public Rectangulo () {// constructor
        alto=20;
        ancho=10;
        relleno=false;
    }
}//fin clase Rectangulo
